В этой папке находятся большие шрифты - для особых случаев


Формат структуры шрифта таков

struct BigFont
{
    int8  height;       // Высота глифа
    uint8 *data;        // Адрес данных
    unt16 offset[256];  // Таблица смещений начала глифов от data. Если == 0xFFFF, то глиф отсутствует в таблице
    uint8 width[256];   // Ширина каждого глифа в битах. При этом каждая строка глифа кратна байту. Т.е. при ширине 9 бит строка занимает два байта. Лишние биты не используются.
};

Скрипт генерирует структуру BigFont и массив c данными шрифта вида

uint8 dataFont24[N] =
{
    233, ...
};

BigFont font24 =
{
    24,
    dataFont24,
    {0, 23, 0, 32, 0xFFFF},
    {23, 12, 32, 0}
};
